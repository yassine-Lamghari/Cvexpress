\newlength{\cvleftcolumnwidth}
\newlength{\cvrightcolumnwidth}
\newlength{\cvcolumngapwidth}

\setlength{\cvleftcolumnwidth}{44mm}
\setlength{\cvcolumngapwidth}{4mm}
\setlength{\cvrightcolumnwidth}{\textwidth-\cvleftcolumnwidth-\cvcolumngapwidth}

\newlength{\cvbeforenameamount}
\newlength{\cvafternameamount}
\newlength{\cvaftersectionskipamount}
\newlength{\cvaftertitleskipamount}
\newlength{\cvafteritemskipamount}
\newlength{\cvbetweensectionandheadingextraskipamount}

\setlength{\cvbeforenameamount}{0mm}
\setlength{\cvafternameamount}{2mm}
\setlength{\cvaftersectionskipamount}{3mm}
\setlength{\cvaftertitleskipamount}{1mm}
\setlength{\cvafteritemskipamount}{2mm}
\setlength{\cvbetweensectionandheadingextraskipamount}{1mm}

\newlength{\cvparskip}
\setlength{\cvparskip}{1mm}

\newcommand{\cvnamestyle}[1]{\Huge\textbf{#1}}
\newcommand{\cvsectionstyle}[1]{\large\textbf{#1}}
\newcommand{\cvtitlestyle}[1]{\large\textbf{#1}}
\newcommand{\cvdurationstyle}[1]{\small#1}
\newcommand{\cvheadingstyle}[1]{\textbf{#1}}


\newcommand{\cvpersonalinfo}[2]{%
    % left and right column
    \begin{minipage}[t]{\cvleftcolumnwidth}
        \vspace{0mm} % XXX hack to align to top
        \raggedleft #1
    \end{minipage}% XXX necessary comment to avoid unwanted space
    \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
    \begin{minipage}[t]{\cvrightcolumnwidth}
        \vspace{0mm} % XXX hack to align to top
        #2
    \end{minipage}

    % space after
    \vspace{\cvafternameamount}
}


\newcommand{\cvname}[1]{%
    % space before
    \vspace{\cvbeforenameamount}

    % name
    \cvnamestyle{#1}

    % space after
    \vspace{\cvparskip}
}

\newcommand{\cvpersonalinfolinewithicon}[2]{%
    % icon, vertically aligned with text
    \makebox[1em][c]{\small#1} 
    % text
    #2

    % space after
    \vspace{\cvparskip}
}


\newcommand{\cvsection}[1]{%
    % left and right column
    \begin{minipage}[t]{\cvleftcolumnwidth}
        \raggedleft\cvsectionstyle{#1}
    \end{minipage}% XXX necessary comment to avoid unwanted space
    \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
    \begin{minipage}[t]{\cvrightcolumnwidth}
        \textcolor{cvrulecolor}{\rule{\cvrightcolumnwidth}{0.3mm}}
    \end{minipage}

    % space after
    \vspace{\cvaftersectionskipamount}
}


\newcommand{\cvitem}[2]{%
    % left and right column
    \begin{minipage}[t]{\cvleftcolumnwidth}
        \raggedleft #1
    \end{minipage}% XXX necessary comment to avoid unwanted space
    \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
    \begin{minipage}[t]{\cvrightcolumnwidth}
        \setlength{\parskip}{\cvparskip} #2
    \end{minipage}\par

    % space after
    \vspace{\cvafteritemskipamount}
}


\newcommand{\cvtitle}[1]{%
    % title
    \cvtitlestyle{#1}

    % space after
    \vspace{\cvaftertitleskipamount}
    % XXX need to subtract cvparskip here
    \vspace{-\cvparskip}
}

\pagestyle{empty}
