\usepackage[empty]{fullpage} % sets margins to one inch and removes headers, footers etc..

\usepackage[normalem]{ulem} % provides "\ul" (uline) command which will break at line breaks

\fancyhf{} % clear all header and footer fields

\renewcommand{\footrulewidth}{0in} % sets thickness of linerule over footer to zero

\raggedbottom{} % makes all pages the height of current page, no extra vertical space added
\raggedright{} % makes all pages the width of current page, no extra horizontal space added

%--------CUSTOM COMMANDS--------------
\newcommand{\resumeItem}[1]{
  \item\small{#1}
}

\newcommand{\resumeItemListStart}{\begin{itemize}[rightmargin=0.11in]}
\newcommand{\resumeItemListEnd}{\end{itemize}}

\newcommand{\resumeSectionType}[3]{
  \item\begin{tabular*}{0.96\textwidth}[t]{
    p{0.15\linewidth}p{0.02\linewidth}p{0.81\linewidth}
  }
    \textbf{#1} & #2 & #3
  \end{tabular*}\vspace{-2pt}
}

\newcommand{\resumeTrioHeading}[3]{
  \item\small{
    \begin{tabular*}{0.96\textwidth}[t]{
      l@{\extracolsep{\fill}}c@{\extracolsep{\fill}}r
    }
      \textbf{#1} & \textit{#2} & #3
    \end{tabular*}
  }
}

\newcommand{\resumeQuadHeading}[4]{
  \item
  \begin{tabular*}{0.96\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{#1} & #2 \\
    \textit{\small#3} & \textit{\small #4} \\
  \end{tabular*}
}

\newcommand{\resumeQuadHeadingChild}[2]{
  \item
  \begin{tabular*}{0.96\textwidth}[t]{l@{\extracolsep{\fill}}r}
    \textbf{\small#1} & {\small#2} \\
  \end{tabular*}
}

\newcommand{\resumeHeadingListStart}{
  \begin{itemize}[leftmargin=0.15in, label={}]
}
\newcommand{\resumeHeadingListEnd}{\end{itemize}}

%%BEGIN_SECTION_SKILLS%%
%--------------SKILLS------------------------
\section{%%SECTION_SKILLS%%}
  \resumeHeadingListStart{}

    \resumeSectionType{Category}{:}{Skill1, Skill2, Skill3}
    %% END REPEAT
  \resumeHeadingListEnd{}

%%END_SECTION_SKILLS%%


%%BEGIN_SECTION_EXPERIENCE%%
%-----------EXPERIENCE-----------------------
\section{%%SECTION_EXPERIENCE%%}
\resumeHeadingListStart{}
  %% REPEAT for each experience entry:
  \resumeQuadHeading{Job Title}{Start -- End}
  {Company}{Location}
    \resumeItemListStart{}
      \resumeItem{Description of achievement or responsibility.}
    \resumeItemListEnd{}
  %% END REPEAT
\resumeHeadingListEnd{}
%---------------------------------------------
%%END_SECTION_EXPERIENCE%%
