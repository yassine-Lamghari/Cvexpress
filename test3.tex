\newlength{\cvparskip}

\newcommand{\cvnamestyle}[1]{\LARGE\textbf{\textcolor{accentcolor}{#1}}}
\newcommand{\cvpersonalinfo}[2]{
    % left and right column
    \begin{minipage}[t]{0.75\textwidth}
        \vspace{0pt}%
        \raggedright
        #2
    \end{minipage}%
    \hfill%
    \begin{minipage}[t]{0.2\textwidth}
        \vspace{0pt}%
        \raggedleft
        #1
    \end{minipage}
    \par\vspace{\cvafternameamount}
}

\newcommand{\cvname}[1]{
    \vspace{\cvbeforenameamount}
    \cvnamestyle{#1}
    \vspace{\cvparskip}
}
\newcommand{\cvpersonalinfolinewithicon}[2]{
    \makebox[1.2em][l]{\textcolor{accentcolor}{#1}}
    #2
    \vspace{\cvparskip}
}
