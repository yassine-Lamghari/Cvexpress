\newcommand{\cvnamestyle}[1]{\LARGE\textbf{\textcolor{accentcolor}{#1}}}

\newcommand{\cvheadingstyle}[1]{\textbf{\textcolor{accentcolor}{#1}}}


\newcommand{\cvpersonalinfo}[2]{
    % left and right column
    \noindent\begin{minipage}[t]{0.75\textwidth}
        \vspace{0mm} % XXX hack to align to top
        \raggedright
        #2
    \end{minipage}%
    \hfill%
    \begin{minipage}[t]{0.2\textwidth}
        \vspace{0mm} % XXX hack to align to top
        \raggedleft
        #1
    \end{minipage}
    \par\vspace{\cvafternameamount}
}


\newcommand{\cvname}[1]{
    % space before
    \vspace{\cvbeforenameamount}

    % name
    \cvnamestyle{#1}

    % space after
    \vspace{\cvparskip}
}

\newcommand{\cvpersonalinfolinewithicon}[2]{
    % icon, vertically aligned with text
    \makebox[1.2em][l]{\textcolor{accentcolor}{#1}} 
    % text
    #2

    % space after
    \vspace{\cvparskip}
}


\newcommand{\cvsection}[1]{
    % left and right column
    \noindent\begin{minipage}[t]{\cvleftcolumnwidth}
        \raggedright\cvsectionstyle{#1}
    \end{minipage}% XXX necessary comment to avoid unwanted space
    \hspace{\cvcolumngapwidth}% XXX necessary comment to avoid unwanted space
    \begin{minipage}[t]{\cvrightcolumnwidth}
        \vspace{1.5mm} % Adjust rule vertical alignment to match text
        \textcolor{cvrulecolor}{\rule{\cvrightcolumnwidth}{0.5mm}}
    \end{minipage}

    % space after
    \vspace{\cvaftersectionskipamount}
}


\newcommand{\cvitem}[2]{
    % left and right column
    \noindent
    \ifx&#1&%
        \begin{minipage}[t]{\textwidth}
            \setlength{\parskip}{\cvparskip} #2
        \end{minipage}\par
    \else
        \begin{minipage}[t]{\cvleftcolumnwidth}
            \raggedright #1
        \end{minipage}%
        \hspace{\cvcolumngapwidth}%
        \begin{minipage}[t]{\cvrightcolumnwidth}
            \setlength{\parskip}{\cvparskip} #2
        \end{minipage}\par
    \fi

    % space after
    \vspace{\cvafteritemskipamount}
}


\newcommand{\cvtitle}[1]{
    % title
    \cvtitlestyle{#1}

    % space after
    \vspace{\cvaftertitleskipamount}
    % XXX need to subtract cvparskip here
    \vspace{-\cvparskip}
}
